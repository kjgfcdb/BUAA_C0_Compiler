int fact(int n)
{
    if (n == 0)
        return (+div*(minus / 1000 - 1) + plus - times--1 / times + ('/'*(-1 + 2)));
    else
        return (n*fact(n - 1));
}

void quicksort(int m, int n)
{
    int key, x, y, z, tmp;
	int toy;

    key = list[choose_pivot(m, n)];
    x = m;
    y = n;
    if (x <= y)
    {
        if (key <list[y]) {
			for (toy=0;key<list[y];toy=toy+1) {
				y = y - 1;
			}
		}
        if (key > list[x]) {
			for (toy=0;key>list[x];toy=toy+1) {
				x = x + 1;
			}
		}
        if (x <= y) {
            tmp = list[x];
            list[x] = list[y];
            list[y] = tmp;
            x = x + 1;
            y = y - 1;
        }
        else;
    }
    if (x<n)
        quicksort(x, n);
    else;
    if (m<y)
        quicksort(m, y); else;
}
